1. At the terminal, Create an empty Git repository under /home/guest/
	$ pwd
	/home/playlabs
	$ mkdir gitPlayLabs
	$ cd gitPlayLabs   
	$ git init gitPlayLabs
	
2. Download from an existing repository
    $ git clone https://github.com/sreenivasd/temp.git

3. Stages the file, ready for commit 
	$ git add application.properties
	
4. Stage all changed files, ready for commit
    $ git add .
	
5. Unstages file, keeping the file changes
    $git reset application.properties 

6. Move or rename a file, a directory
    $git mv application.properties app.properties

7. Remove files from the working tree and from the index
    $git rm application.properties
	
8. Show full change history
	$git log
	
9. Print lines matching a pattern
    $git log -p | grep app  

10. List new or modified files not yet committed
    $git status
	
11. Show the file changes for a commit id and/or file
    $git show d76481136ad8d036e19e854082993c62f61e6f85
	
12. Record changes to the repository
    $ggit commit -m "first commitâ€œ
	
13. List the change dates and authors for a file 
    $git blame -l app.properties
	
14. Show the changes to staged files
	$git diff --cached
	
15. List all local branches    
	$git branch

16. Create a new branch called new_branch
	$git branch new_branch

17. Merge branch_a into branch_b
	$git checkout branch_b

18. Tag the current commit
	$git tag tag_current

19. Join two or more development histories together
	$git merge

20 Get the latest changes from origin  
	$git fetch

21. Fetch the latest changes from origin and merge 
	$git pull

22. Push local changes to the origin 
	$git push -u origin master
